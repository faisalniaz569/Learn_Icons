Example 1:

<!DOCTYPE html>
<html lang="en">

<head>
    <meta charset="UTF-8">
    <title>fontawesome icon 4</title>
    <link rel="stylesheet" href="font-awesome-4.7.0/css/font-awesome.min.css">
</head>

<body>

    <h1>fa fa-address-book</h1>

    <i class="fa fa-address-book"></i>
    <i class="fa fa-address-book" style="font-size:24px"></i>
    <i class="fa fa-address-book" style="font-size:36px"></i>
    <i class="fa fa-address-book" style="font-size:48px;color:red"></i>
    <br>

    <p>Used on a button:</p>
    <button style="font-size:24px">Button <i class="fa fa-address-book"></i></button>

    <p>Unicode:</p>
    <i style="font-size:24px" class="fa">&#xf2b9;</i>


</body>

</html>
